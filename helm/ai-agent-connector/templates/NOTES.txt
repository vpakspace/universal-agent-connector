1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "ai-agent-connector.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "ai-agent-connector.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "ai-agent-connector.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  echo "Access the application via port-forward:"
  echo "  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "ai-agent-connector.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}"
  echo "  Then access http://localhost:{{ .Values.service.port }}"
{{- end }}

2. Health check endpoint:
   {{- if .Values.ingress.enabled }}
   http{{ if .Values.ingress.tls }}s{{ end }}://{{ (index .Values.ingress.hosts 0).host }}/api/health
   {{- else }}
   kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "ai-agent-connector.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}
   Then access http://localhost:{{ .Values.service.port }}/api/health
   {{- end }}

3. Check pod status:
   kubectl get pods --namespace {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "ai-agent-connector.name" . }}

4. View logs:
   kubectl logs --namespace {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "ai-agent-connector.name" . }} --tail=100

5. {{- if .Values.autoscaling.enabled }}
   Auto-scaling is enabled ({{ .Values.autoscaling.minReplicas }}-{{ .Values.autoscaling.maxReplicas }} replicas)
   Check HPA status:
   kubectl get hpa --namespace {{ .Release.Namespace }} {{ include "ai-agent-connector.fullname" . }}
   {{- else }}
   Auto-scaling is disabled. Current replicas: {{ .Values.replicaCount }}
   {{- end }}
